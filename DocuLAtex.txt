\section{Pseudocode}
\begin{algorithm}[H]
\caption{formIsValid}\label{alg:formIsValid}
\begin{algorithmic}
\Require $name, phoneNumber,day,month,year$
\Ensure nameIsValid \&\& nameIsValid \&\& dateIsValid
\State return nameIsValid \&\& nameIsValid \&\& dateIsValid
\end{algorithmic}
\end{algorithm}
\begin{algorithm}[H]
\caption{nameIsValid}\label{alg:nameIsValid}
\begin{algorithmic}
\Require $name$
\Ensure $50 \geq name.length \geq 1$
\If{name is empty or $name.length \geq 50$}
    \State return false
\Else{}
    \State return true
\EndIf
\end{algorithmic}
\end{algorithm}
\begin{algorithm}[H]
\caption{phoneNumberIsValid}\label{alg:phoneNumberIsValid}
\begin{algorithmic}
\Require $number$
\Ensure $optionalPrefix\{0,15\}+[0-9]\{8\}$
\State $prefix \gets getOptionalPrefix(number)$
\State $suffix \gets removePrefix(number)$
\State $rawSuffix \gets removeFillers(suffix)$
\If{$prefix.length>MAXE164$}
    \State return false
\EndIf
\If{$not(rawSuffix.length==8)$}
    \State return false
\EndIf
\If{$isNotDigit(rawSuffix)$}
    \State return false
\EndIf
\end{algorithmic}
\end{algorithm}
\begin{algorithm}[H]
\caption{dateIsValid}\label{alg:dateIsValid}
\begin{algorithmic}
\Require $date,month,year$
\Ensure yearIsValid \&\& dayIsValid
\State $dayIsValid \gets true$
\If{$0 \geq year \geq MAXINT$}
    \State return false
\EndIf
\If{$date<1$}
    \State return false
\EndIf
\If{month in ${1,3,5,7,8,10,12}$}
    \State return $31\geq date$
\ElsIf{month in ${4,6,9,11}$}
    \State return $30\geq date$
\Else{}
    \If{$year\%4!=0$}
    \State return $29\geq date$
    \Else{}
    \State return $28\geq date$
    \EndIf
\EndIf
\end{algorithmic}
\end{algorithm}
